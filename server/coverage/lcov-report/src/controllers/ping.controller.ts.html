import express, { NextFunction, Request, Response, Router } from "express";
class PingController {
    getRoutes(): Router {
        let router = express.Router();
        // '/api/users/:id'
        router.get('/', [], (req: Request, res: Response, next: NextFunction) => {
            this.get(req, res, next);
        });
        return router;
    }
